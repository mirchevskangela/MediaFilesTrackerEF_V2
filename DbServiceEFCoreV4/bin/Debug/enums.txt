using System;
enum DaysOfWeek
{
    Monday,
    Tuesday,
    Wednesday,
    Thursday,
    Friday,
    Saturday,
    Sunday
}
public enum ActionsEnum
{
    Created = 1,
    Modified = 2,
    Deleted = 3
}
class HelloWorld {
  static void Main() {
    Console.WriteLine("Hello World");
    // Using enum values
        DaysOfWeek today = DaysOfWeek.Monday;
        Console.WriteLine("Today is " + today);

        // Enum iteration
        foreach (DaysOfWeek day in Enum.GetValues(typeof(DaysOfWeek)))
        {
            Console.WriteLine(day);
        }

        // Enum parsing
        string userInput = "Fridvfvfdzbay";
        DaysOfWeek parsedDay;
        if (Enum.TryParse(userInput, out parsedDay))
        {
            Console.WriteLine("Parsed day: " + 	parsedDay);
        }
        else
        {
            Console.WriteLine("Invalid input");
        }
        int createdValue = (int)ActionsEnum.Created;


        Console.WriteLine("Created value: " + createdValue);
  }
}
